### 1.3 业主报告应包含的核心内容（行业标准）

根据PropertyCEO、NARPM、IREM等行业标准，一份完整的月度业主报告应包含：

**Section 1: Financial Summary**
- Rent collected (amount, date, payment method)
- Expenses (itemized with vendor names, amounts, categories)
- Management fee (clearly broken out)
- Net disbursement (what owner received)
- Year-to-date totals
- Reserve balance

**Section 2: Property Status**
- Occupancy status (occupied/vacant/turnover)
- Lease expiration date
- Current monthly rent
- Tenant payment history

**Section 3: Maintenance Activity**
- Work orders completed (what, when, cost)
- Open work orders (pending + why)
- Upcoming maintenance
- Before/after photos

**Section 4: Market Update (Optional)**
- Local rent trends
- Neighborhood activity
- Renewal recommendation

### 3.2 报告生成引擎

```swift
import Foundation
import SwiftData

struct MonthlyReport {
    let property: Property
    let period: DateInterval
    let income: [Transaction]
    let expenses: [Transaction]
    let maintenanceOrders: [MaintenanceOrder]
    let tenants: [Tenant]
    
    var totalIncome: Double {
        income.filter { $0.transactionType == .income }.reduce(0) { $0 + $1.amount }
    }
    
    var totalExpenses: Double {
        expenses.filter { $0.transactionType == .expense }.reduce(0) { $0 + $1.amount }
    }
    
    var managementFee: Double {
        totalIncome * (property.managementFeePercent / 100.0)
    }
    
    var netIncome: Double {
        totalIncome - totalExpenses - managementFee
    }
    
    var occupancyRate: Double {
        let totalUnits = Double(property.units)
        let occupiedUnits = Double(tenants.filter { $0.paymentStatus != .vacant }.count)
        return totalUnits > 0 ? (occupiedUnits / totalUnits) * 100 : 0
    }
    
    var expensesByCategory: [TransactionCategory: Double] {
        Dictionary(grouping: expenses.filter { $0.transactionType == .expense }, by: { $0.category })
            .mapValues { $0.reduce(0) { $0 + $1.amount } }
    }
    
    var yearToDateIncome: Double {
        let calendar = Calendar.current
        let startOfYear = calendar.date(from: calendar.dateComponents([.year], from: period.start))!
        let ytdInterval = DateInterval(start: startOfYear, end: period.end)
        return income.filter { ytdInterval.contains($0.date) && $0.transactionType == .income }
            .reduce(0) { $0 + $1.amount }
    }
    
    var yearToDateExpenses: Double {
        let calendar = Calendar.current
        let startOfYear = calendar.date(from: calendar.dateComponents([.year], from: period.start))!
        let ytdInterval = DateInterval(start: startOfYear, end: period.end)
        return expenses.filter { ytdInterval.contains($0.date) && $0.transactionType == .expense }
            .reduce(0) { $0 + $1.amount }
    }
}

@MainActor
final class ReportGenerator: ObservableObject {
    private var modelContext: ModelContext
    
    init(modelContext: ModelContext) {
        self.modelContext = modelContext
    }
    
    func generateMonthlyReport(for property: Property, month: Date) throws -> MonthlyReport {
        let calendar = Calendar.current
        let startOfMonth = calendar.date(from: calendar.dateComponents([.year, .month], from: month))!
        let endOfMonth = calendar.date(byAdding: DateComponents(month: 1, day: -1), to: startOfMonth)!
        let period = DateInterval(start: startOfMonth, end: endOfMonth)
        
        let allTransactions = property.transactions.filter { period.contains($0.date) }
        let income = allTransactions.filter { $0.transactionType == .income }
        let expenses = allTransactions.filter { $0.transactionType == .expense }
        
        let maintenance = property.maintenanceOrders.filter { order in
            if let completed = order.completedAt {
                return period.contains(completed) || period.contains(order.createdAt)
            }
            return period.contains(order.createdAt)
        }
        
        return MonthlyReport(
            property: property,
            period: period,
            income: income,
            expenses: expenses,
            maintenanceOrders: maintenance,
            tenants: property.tenants
        )
    }
    
    func generatePortfolioReport(month: Date) throws -> [MonthlyReport] {
        let descriptor = FetchDescriptor<Property>()
        let properties = try modelContext.fetch(descriptor)
        return properties.map { try? generateMonthlyReport(for: $0, month: month) }
            .compactMap { $0 }
    }
}
```

```swift
import PDFKit

final class PDFReportGenerator {
    static func generateOwnerReport(_ report: MonthlyReport, branding: BrandingConfig) -> Data {
        let pageWidth: CGFloat = 612
        let pageHeight: CGFloat = 792
        let margin: CGFloat = 50
        let pdfMetaData = [
            kCGPDFContextTitle: "Monthly Owner Report - \(report.property.name)",
            kCGPDFContextCreator: "RentBrief"
        ]
        let format = UIGraphicsPDFRendererFormat()
        format.documentInfo = pdfMetaData as [String: Any]
        
        let renderer = UIGraphicsPDFRenderer(bounds: CGRect(x: 0, y: 0, width: pageWidth, height: pageHeight), format: format)
        
        let data = renderer.pdfData { context in
            context.beginPage()
            var y: CGFloat = margin
            
            y = drawHeader(y: y, margin: margin, pageWidth: pageWidth, report: report, branding: branding, context: context.cgContext)
            y = drawFinancialSummary(y: y, margin: margin, pageWidth: pageWidth, report: report, context: context.cgContext)
            y = drawPropertyStatus(y: y, margin: margin, pageWidth: pageWidth, report: report, context: context.cgContext)
            y = drawExpenseBreakdown(y: y, margin: margin, pageWidth: pageWidth, report: report, context: context.cgContext)
            
            if y > pageHeight - 200 {
                context.beginPage()
                y = margin
            }
            y = drawMaintenanceSummary(y: y, margin: margin, pageWidth: pageWidth, report: report, context: context.cgContext)
        }
        
        return data
    }
    
    private static func drawHeader(y: CGFloat, margin: CGFloat, pageWidth: CGFloat, report: MonthlyReport, branding: BrandingConfig, context: CGContext) -> CGFloat {
        var currentY = y
        
        let dateFormatter = DateFormatter()
        dateFormatter.dateFormat = "MMMM yyyy"
        let periodString = dateFormatter.string(from: report.period.start)
        
        let titleAttributes: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 24, weight: .bold),
            .foregroundColor: branding.primaryColor
        ]
        let titleString = "Monthly Owner Report"
        titleString.draw(at: CGPoint(x: margin, y: currentY), withAttributes: titleAttributes)
        currentY += 35
        
        let subtitleAttributes: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 14, weight: .medium),
            .foregroundColor: UIColor.darkGray
        ]
        "\(report.property.name) — \(periodString)".draw(at: CGPoint(x: margin, y: currentY), withAttributes: subtitleAttributes)
        currentY += 25
        
        let detailAttributes: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 11),
            .foregroundColor: UIColor.gray
        ]
        "\(report.property.address), \(report.property.city), \(report.property.state) \(report.property.zipCode)".draw(at: CGPoint(x: margin, y: currentY), withAttributes: detailAttributes)
        currentY += 20
        "Owner: \(report.property.ownerName)".draw(at: CGPoint(x: margin, y: currentY), withAttributes: detailAttributes)
        currentY += 30
        
        context.setStrokeColor(branding.primaryColor.cgColor)
        context.setLineWidth(2)
        context.move(to: CGPoint(x: margin, y: currentY))
        context.addLine(to: CGPoint(x: pageWidth - margin, y: currentY))
        context.strokePath()
        currentY += 20
        
        return currentY
    }
    
    private static func drawFinancialSummary(y: CGFloat, margin: CGFloat, pageWidth: CGFloat, report: MonthlyReport, context: CGContext) -> CGFloat {
        var currentY = y
        
        let sectionTitleAttributes: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 16, weight: .semibold),
            .foregroundColor: UIColor.darkGray
        ]
        let valueAttributes: [NSAttributedString.Key: Any] = [
            .font: UIFont.monospacedDigitSystemFont(ofSize: 14, weight: .medium),
            .foregroundColor: UIColor.black
        ]
        let labelAttributes: [NSAttributedString.Key: Any] = [
            .font: UIFont.systemFont(ofSize: 13),
            .foregroundColor: UIColor.gray
        ]
        
        "Financial Summary".draw(at: CGPoint(x: margin, y: currentY), withAttributes: sectionTitleAttributes)
        currentY += 30
        
        let items: [(String, Double, Bool)] = [
            ("Total Income", report.totalIncome, true),
            ("Total Expenses", report.totalExpenses, false),
            ("Management Fee (\(String(format: "%.0f", report.property.managementFeePercent))%)", report.managementFee, false),
            ("Net Income", report.netIncome, true)
        ]
        
        for (label, amount, isHighlight) in items {
            let attrs = isHighlight ? [
                .font: UIFont.monospacedDigitSystemFont(ofSize: 16, weight: .bold),
                .foregroundColor: amount >= 0 ? UIColor.systemGreen : UIColor.systemRed
            ] : valueAttributes
            
            label.draw(at: CGPoint(x: margin, y: currentY), withAttributes: labelAttributes)
            let amountString = String(format: "$%,.2f", amount)
            let amountSize = amountString.size(withAttributes: attrs)
            amountString.draw(at: CGPoint(x: pageWidth - margin - amountSize.width, y: currentY), withAttributes: attrs)
            currentY += 25
        }
        
        currentY += 10

    }
    
    private static func drawPropertyStatus(y: CGFloat, margin: CGFloat, pageWidth: CGFloat, report: MonthlyReport, context: CGContext) -> CGFloat {
```

## 八、符合美国市场的UI设计

### 8.1 设计原则

| 原则 | 说明 | 实现方式 |
|------|------|----------|
| **清晰优先** | 美国用户重视效率，信息一目了然 | 大数字KPI卡片，关键指标突出 |
| **信任感** | 涉及财务数据，UI必须专业可靠 | 蓝色主色调，银行级UI质感 |
| **移动优先** | PM在物业现场需要手机操作 | 底部Tab导航，大触摸目标 |
| **深色模式** | 美国用户深色模式使用率高 | 原生支持，自动切换 |
| **无障碍** | ADA合规要求 | VoiceOver、Dynamic Type支持 |
